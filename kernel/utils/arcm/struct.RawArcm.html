impl<T, U, const INT: bool> CoerceUnsized<RawArcm<U, INT>> for RawArcm<T, INT>where T: Unsize<U> + ?Sized, U: ?Sized,

Allow type coercion from RawArcm<T> to RawArcm<U> where U is Unsized and coercion exist from T to U This is used to store dyn Trait from a struct that implement the trait for example

Examples

trait Foo {
    fn bar(&self) -> u32;
}
struct Foobar {}
impl Foo for Foobar {
    fn bar(&self) -> u32 { 42 }
}
fn exmaple() {
    let foo: Foobar = Foobar{};
    let arcm: RawArcm<dyn Foo, false> = RawArcm::new(foo);
    assert_eq!(arcm.lock().bar(), 42);
}
